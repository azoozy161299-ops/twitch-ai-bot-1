// بوت Twitch ذكي يربط Nightbot بـ ChatGPT عبر OpenAI API

import express from "express";
import dotenv from "dotenv";
import { Configuration, OpenAIApi } from "openai";

// يقرأ ملف .env لتخزين المفتاح بشكل آمن
dotenv.config();

const app = express();

// تمكين قراءة بيانات POST و URL-encoded
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// إعداد OpenAI API باستخدام المفتاح من .env
const configuration = new Configuration({
  apiKey: process.env.OPENAI_API_KEY
});
const openai = new OpenAIApi(configuration);

// Endpoint للاختبار البسيط: يرد دائمًا برسالة جاهزة
app.get("/", (req, res) => {
  res.send("✅ Bot is online and ready!");
});

// Endpoint للرد الذكي على رسائل Nightbot
app.get("/reply", async (req, res) => {
  const userMessage = req.query.message || "مرحبا"; // الرسالة التي يكتبها المشاهد

  try {
    // استدعاء ChatGPT للرد على الرسالة
    const completion = await openai.createChatCompletion({
      model: "gpt-3.5-turbo", // يمكنك تغييره إلى gpt-4 إذا متاح لك
      messages: [{ role: "user", content: userMessage }]
    });

    // الحصول على نص الرد من ChatGPT
    const reply = completion.data.choices[0].message.content;

    // إرسال الرد إلى Nightbot ليظهر في الشات
    res.send(reply);

  } catch (error) {
    console.error(error);
    // رسالة احتياطية في حالة حدوث خطأ
    res.send("حدث خطأ أثناء محاولة الرد 😢");
  }
});

// تشغيل السيرفر على البورت المحدد في Render أو 3000 محليًا
const PORT = process.env.PORT || 3000;
app.listen(PORT, () => console.log(`🚀 Bot running on port ${PORT}`));
